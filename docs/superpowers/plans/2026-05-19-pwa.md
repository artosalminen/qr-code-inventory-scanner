# PWA Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Make the QR inventory app installable (add to home screen) and capable of queuing scans while offline, syncing automatically on reconnect.

**Architecture:** Web app manifest + minimal service worker for installability; IndexedDB-backed scan queue for offline operation; scanner forks on `navigator.onLine` to enqueue instead of POST when offline; `_app.tsx` flushes the queue on mount and on reconnect, showing a sync result notification.

**Tech Stack:** Next.js 16 Pages Router, React 19, TypeScript, IndexedDB (browser native), `sharp` (icon generation, already installed), Jest + jsdom + @testing-library/react (existing test setup)

`src/lib/scan-queue.ts`:

export interface QueuedScan {
  id: string;
  qrCode: string;
  projectId: string;
  action: string;
  condition: string;
  notes: string;
  timestamp: number;
}

const DB_NAME = 'qr-inventory';
const STORE = 'pending_scans';
const DB_VERSION = 1;

function openDb(): Promise<IDBDatabase> {
  return new Promise((resolve, reject) => {
    const req = indexedDB.open(DB_NAME, DB_VERSION);
    req.onupgradeneeded = () => {
      if (!req.result.objectStoreNames.contains(STORE)) {
        req.result.createObjectStore(STORE, { keyPath: 'id' });
      }
    };
    req.onsuccess = () => resolve(req.result);
    req.onerror = () => reject(req.error);
  });
}

export async function enqueue(scan: QueuedScan): Promise<void> {
  const db = await openDb();
  return new Promise((resolve, reject) => {
    const tx = db.transaction(STORE, 'readwrite');
    tx.objectStore(STORE).add(scan);
    tx.oncomplete = () => resolve();
    tx.onerror = () => reject(tx.error);
  });
}

export async function getAll(): Promise<QueuedScan[]> {
  const db = await openDb();
  return new Promise((resolve, reject) => {
    const tx = db.transaction(STORE, 'readonly');
    const req = tx.objectStore(STORE).getAll();
    req.onsuccess = () =>
      resolve((req.result as QueuedScan[]).sort((a, b) => a.timestamp - b.timestamp));
    req.onerror = () => reject(req.error);
  });
}

export async function remove(id: string): Promise<void> {
  const db = await openDb();
  return new Promise((resolve, reject) => {
    const tx = db.transaction(STORE, 'readwrite');
    tx.objectStore(STORE).delete(id);
    tx.oncomplete = () => resolve();
    tx.onerror = () => reject(tx.error);
  });
}
```

import { useState, useEffect } from 'react';

export function useOnlineStatus(): boolean {
  const [isOnline, setIsOnline] = useState<boolean>(
    typeof navigator !== 'undefined' ? navigator.onLine : true,
  );

  useEffect(() => {
    const handleOnline = () => setIsOnline(true);
    const handleOffline = () => setIsOnline(false);
    window.addEventListener('online', handleOnline);
    window.addEventListener('offline', handleOffline);
    return () => {
      window.removeEventListener('online', handleOnline);
      window.removeEventListener('offline', handleOffline);
    };
  }, []);

  return isOnline;
}
```

`src/hooks/useQueueFlush.ts`:
```ts
import { useEffect, useRef } from 'react';
import axios from 'axios';
import { getAll, remove } from '@/lib/scan-queue';

export interface FlushResult {
  succeeded: number;
  failed: Array<{ qrCode: string; error: string }>;
}

async function flushQueue(): Promise<FlushResult> {
  const items = await getAll();
  const result: FlushResult = { succeeded: 0, failed: [] };

  for (const item of items) {
    try {
      await axios.post('/api/boxes/scan', {
        projectId: item.projectId,
        qrCode: item.qrCode,
        action: item.action,
        condition: item.condition,
        notes: item.notes,
      });
      result.succeeded++;
    } catch (error: any) {
      result.failed.push({
        qrCode: item.qrCode,
        error: error.response?.data?.error ?? 'Sync failed',
      });
    }
    await remove(item.id);
  }

  return result;
}

export function useQueueFlush(onResult?: (result: FlushResult) => void): void {
  const onResultRef = useRef(onResult);
  onResultRef.current = onResult;

  useEffect(() => {
    const run = async () => {
      const result = await flushQueue();
      if (result.succeeded > 0 || result.failed.length > 0) {
        onResultRef.current?.(result);
      }
    };

    run();
    window.addEventListener('online', run);
    return () => window.removeEventListener('online', run);
  }, []);
}
```
